/* Em caso de Emergencia */

CREATE TABLE IF NOT EXISTS Produto (
id SERIAL PRIMARY KEY,
nome VARCHAR(30) NOT NULL,
preco NUMERIC(10, 2) NOT NULL
);

CREATE TABLE IF NOT EXISTS Cliente (
id SERIAL PRIMARY KEY,
nome VARCHAR(30) NOT NULL,
email VARCHAR(30) UNIQUE NOT NULL,
senha VARCHAR(30) NOT NULL,
telefone VARCHAR(30),
data_nasc VARCHAR(30),
id_endereco_principal INT,

FOREIGN KEY (id_endereco_principal) REFERENCES endereco (id) ON DELETE SET NULL
);

CREATE TABLE IF NOT EXISTS Endereco (
id SERIAL PRIMARY KEY,
rua VARCHAR(30) NOT NULL,
numero VARCHAR(30) NOT NULL,
cidade VARCHAR(30) NOT NULL,
bairro VARCHAR(30) NOT NULL,
cep VARCHAR(30),
complemento VARCHAR(30)
);

CREATE TABLE IF NOT EXISTS Veiculo (
id SERIAL PRIMARY KEY,
placa VARCHAR(30) UNIQUE NOT NULL,
ano VARCHAR(30),
modelo VARCHAR(30)
);

CREATE TABLE IF NOT EXISTS Entregador (
id SERIAL PRIMARY KEY,
nome VARCHAR(30) NOT NULL ,
telefone VARCHAR(30),
data_nasc VARCHAR(30),
email VARCHAR(30) UNIQUE NOT NULL,
senha VARCHAR(30) NOT NULL,

id_veiculo INTEGER UNIQUE,
FOREIGN KEY (id_veiculo) REFERENCES veiculo(id)
);

CREATE TABLE IF NOT EXISTS Pedido (
id SERIAL PRIMARY KEY,

id_cliente INTEGER NOT NULL,
id_endereco INTEGER NOT NULL,

status VARCHAR(30) NOT NULL,

valor_total NUMERIC(10, 2) NOT NULL,
forma_pagamento VARCHAR(30),

FOREIGN KEY (id_cliente) REFERENCES Cliente(id),
FOREIGN KEY (id_endereco) REFERENCES Endereco(id)
);

CREATE TABLE ItensPedido (
id SERIAL PRIMARY KEY,

id_pedido INTEGER NOT NULL,
id_produto INTEGER NOT NULL,

quantidade INTEGER NOT NULL,

preco_unitario_historico NUMERIC(10, 2) NOT NULL,

FOREIGN KEY (id_pedido) REFERENCES Pedido(id),
FOREIGN KEY (id_produto) REFERENCES Produto(id)
);
